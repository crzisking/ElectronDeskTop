## 2. Pinia Store 設計

### 2.1 `config.store.ts`

```typescript
import { defineStore } from 'pinia'
import type { AppConfig, SidebarItem, QuickMenuItem, SystemLink } from '@/types/config.types'

export const useConfigStore = defineStore('config', {
  state: () => ({
    appConfig: null as AppConfig | null,
    isLoaded: false,
  }),

  getters: {
    sidebarItems: (state): SidebarItem[] =>
      state.appConfig?.sidebar.items.filter(i => i.enabled) ?? [],

    floatingBallMenuItems: (state): QuickMenuItem[] =>
      state.appConfig?.floatingBall.quickMenu.filter(i => i.enabled) ?? [],

    platformSystems: (state): SystemLink[] =>
      state.appConfig?.unifiedPlatform.systems ?? [],

    aiConfig: (state) => state.appConfig?.aiQuickFunctions,

    contactConfig: (state) => state.appConfig?.quickContact,
  },

  actions: {
    async loadConfig() {
      const config = await window.electronAPI.config.read() as AppConfig
      this.appConfig = config
      this.isLoaded = true
    },
  }
})
```

## 3. API 層設計

### 3.1 HTTP Client 工廠

```typescript
// src/api/http-client.ts
import axios, { type AxiosInstance } from 'axios'
import { setupAuthInterceptor } from './interceptors/auth.interceptor'
import { setupErrorInterceptor } from './interceptors/error.interceptor'

export function createHttpClient(baseURL: string): AxiosInstance {
  const client = axios.create({
    baseURL,
    timeout: 15000,
    headers: { 'Content-Type': 'application/json' }
  })

  setupAuthInterceptor(client)
  setupErrorInterceptor(client)

  return client
}
```

### 3.2 Auth 攔截器

```typescript
// src/api/interceptors/auth.interceptor.ts
import type { AxiosInstance } from 'axios'
import { useAuthStore } from '@/stores/auth.store'

export function setupAuthInterceptor(client: AxiosInstance) {
  client.interceptors.request.use((config) => {
    const authStore = useAuthStore()
    if (authStore.accessToken) {
      config.headers.Authorization = `Bearer ${authStore.accessToken}`
    }
    return config
  })

  // 401 處理（token 失效）
  client.interceptors.response.use(
    (response) => response,
    async (error) => {
      if (error.response?.status === 401) {
        const authStore = useAuthStore()
        await authStore.logout()
        // router.push({ name: 'login' })  // TODO: 登錄功能完成後啟用
      }
      return Promise.reject(error)
    }
  )
}
```

### 3.3 統一錯誤類型

```typescript
// src/types/api.types.ts
export interface ApiError {
  code: string
  message: string
  statusCode: number
  details?: unknown
}

// 在 error.interceptor.ts 中將 Axios 錯誤標準化為 ApiError
export function normalizeError(error: unknown): ApiError {
  if (axios.isAxiosError(error)) {
    return {
      code: error.response?.data?.code ?? 'UNKNOWN',
      message: error.response?.data?.message ?? error.message,
      statusCode: error.response?.status ?? 0,
      details: error.response?.data,
    }
  }
  return { code: 'CLIENT_ERROR', message: String(error), statusCode: 0 }
}
```

## 4. TypeScript 規範

### 4.1 強制規則

| 規則 | 說明 |
|------|------|
| 禁止 `any` | 所有類型必須明確，未知類型用 `unknown` 替代 |
| 嚴格 null 檢查 | 所有可能為 null/undefined 的地方必須顯式處理 |
| 接口優先 | 數據結構用 `interface`，工具類型用 `type` |
| 枚舉避用 | 用 `const` 對象 + `typeof` 替代 `enum`（防止運行時問題）|
| 導入類型 | 僅用於類型的導入使用 `import type` |

### 4.2 類型聲明位置

```
src/types/
├── config.types.ts     → AppConfig 及其子類型
├── api.types.ts        → 所有 API 請求/響應類型、ApiError
├── ipc-channels.ts     → IPC 頻道常量（渲染進程端）
└── electron.d.ts       → window.electronAPI 類型聲明（擴展 Window 接口）
```

```typescript
// src/types/electron.d.ts
export {}
declare global {
  interface Window {
    electronAPI: {
      config: {
        read: () => Promise<AppConfig>
        write: (config: Partial<AppConfig>) => Promise<void>
      }
      window: {
        minimize: () => void
        maximize: () => void
        close: () => void
        hide: () => void
      }
      floatingBall: {
        show: () => void
        hide: () => void
        startDrag: () => void
        stopDrag: () => void
        onMenuAction: (cb: (actionId: string) => void) => void
      }
      auth: {
        getToken: () => Promise<string | null>
        setToken: (token: string) => Promise<void>
        deleteToken: () => Promise<void>
      }
      on: (channel: string, cb: (...args: unknown[]) => void) => void
      off: (channel: string, cb: (...args: unknown[]) => void) => void
    }
  }
}
```


## 8. 開發啟動順序

首次搭建項目的步驟：

```bash

npm create @quick-start/electron enterprise-desktop-client -- --template vue-ts

# 2. 安裝依賴
cd enterprise-desktop-client
npm install

# 3. 安裝業務依賴
npm install element-plus @element-plus/icons-vue pinia vue-router axios electron-store keytar

# 4. 安裝開發依賴
npm install -D unplugin-auto-import unplugin-vue-components

# 5. 按文檔目錄結構組織代碼

# 6. 啟動開發
npm run dev
```
